CREATE DATABASE ARTMUSEUM;
CREATE TABLE ART_OBJECT
	(IDno		VARCHAR(20)		NOT NULL,
	 Year		DATE,
	 Title		VARCHAR(30),
	 Description	VARCHAR(100),
	 PRIMARY KEY(IDno),
	 UNIQUE(IDno),
CREATE TABLE PAINTING 
	(PaintType	CHAR(10)		NOT NULL,
	 Material	CHAR(15),
	 Style		CHAR(15),
	 IDno		VARCHAR(20)		NOT NULL;
	 PRIMARY KEY(IDno),
	 FOREIGN KEY(IDno) REFERENCES ART_OBJECT(IDno));
CREATE TABLE SCULPTURE
	(Material	CHAR(10),
	 Height		FLOAT			NOT NULL,
	 Weight 	FLOAT			NOT NULL,
	 Style		CHAR(15),
	 IDno		VARCHAR(20)		NOT NULL;
	 PRIMARY KEY(IDno),
	 FOREIGN KEY(IDno) REFERENCES ART_OBJECT(IDno));
CREATE TABLE STATUE
	(Honoree	VARCHAR(15),
	 IDno		VARCHAR(20)		NOT NULL,
	 PRIMARY KEY(IDno),
	 FOREIGN KEY(IDno) REFERENCES ART_OBJECT(IDno));
CREATE TABLE OTHER
	(Type 		CHAR(15),
	 Style 		CHAR(15),
	 IDno		VARCHAR(20)		NOT NULL;
	 PRIMARY KEY(IDno),
	 FOREIGN KEY(IDno) REFERENCES ART_OBJECT(IDno));
CREATE TABLE ARTIST
	(Epoch		CHAR(15) 		NOT NULL,	
	 MainStyle	CHAR(15)	
	 Country	CHAR(20)	
	 AName		CHAR(30)		NOT NULL,
	 Born		DATE			NOT NULL,
	 Died		DATE,
	 IDno		VARCHAR(20)		NOT NULL,
	 PRIMARY KEY(AName),
	 UNIQUE(AName),
	 FOREIGN KEY(IDno) REFERENCES ART_OBJECT(IDno));
CREATE TABLE EXHIBITION
	(EName		CHAR(30)		NOT NULL,
	 StartDate	DATE,
	 EndDate	DATE,
	 AName		CHAR(30)		NOT NULL,
	 PRIMARY KEY(AName),
	 UNIQUE(EName),
	 FOREIGN KEY(AName) REFERENCES ARTIST(AName));
CREATE TABLE PERMANENT 
	(Cost		FLOAT			NOT NULL,
	 DateAcquired	DATE,
	 onDisplay_Y/n?	CHAR(1),
	 Source		VARCHAR(30)		NOT NULL,
	 IDno		VARCHAR(20)		NOT NULL,
	 PRIMARY KEY(IDno),
	 FOREIGN KEY(IDno) REFERENCES ART_OBJECT(IDno));
CREATE TABLE BORROWED
	(DateBorrowed	DATE,
	 DateToReturn	DATE,
	 IDno		VARCHAR(20)		NOT NULL,
	 PRIMARY KEY(IDno),
	 FOREIGN KEY(IDno) REFERENCES ART_OBJECT(IDno));
CREATE TABLE COLLECTION
	(ColName	CHAR(30)		NOT NULL,
	 Description	CHAR(30),
	 ContactPerson	CHAR(30),
	 Type			CHAR(10),
	 PhoneNumber	VARCHAR(10),
	 IDno		VARCHAR(20)		NOT NULL,
	 PRIMARY KEY(IDno),
	 UNIQUE(ColName),
	 FOREIGN KEY(IDno) REFERENCES ART_OBJECT(IDno));
