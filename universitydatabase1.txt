CREATE DATABASE universityDatabase;
CREATE TABLE STUDENT
	(FN			VARCHAR(15)			NOT NULL,
	 MN			VARCHAR(15)			NOT NULL,
	 LN			VARCHAR(15)			NOT NULL,
	 SNum		CHAR(9)				NOT NULL,
	 SSN		CHAR(9)				NOT NULL,
	 BirthDate	DATE,
	 Sex		CHAR,
	 class		VARCHAR(10),
	 StreetAdrs	VARCHAR(30),
	 city		VARCHAR(30),
	 State		VARCHAR(30),
	 zip		VARCHAR(6),
	 CPhone		INT					NOT NULL,
	 PAddress 	VARCHAR(30),
	 PPhone		INT					NOT NULL,
	 DegreeProgram	VARCHAR(10)		NOT NULL,
	PRIMARY KEY(SNum),
	UNIQUE(SNum),
	UNIQUE(SSN),
CREATE TABLE DEPARTMENT
	(Dname		VARCHAR(15)			NOT NULL,
	 DCode		VARCHAR(4)			NOT NULL,
	 DOffice	VARCHAR(30)			NOT NULL,
	 DPhone		INT					NOT NULL,
	 College 	VARCHAR(30)			NOT NULL,
	 SNum		CHAR(9)				NOT NULL,
	 PRIMARY KEY(DCode),
	 UNIQUE(DName),
	 FOREIGN KEY(SNum) REFERENCES STUDENT(SNum));
CREATE TABLE COURSE
	(CName 		CHAR(10)			NOT NULL,
	 CNum		CHAR(10)			NOT NULL,
	 CDescription	VARCHAR(30)		NOT NULL,
	 SemesterHours	INT				NOT NULL,
	 Level		VARCHAR(10)			NOT NULL,
	 DCode		VARCHAR(4)			NOT NULL,
	 PRIMARY KEY(CNum),
	 UNIQUE(CNum),
	 FOREIGN KEY(DCode) REFERENCES DEPARTMENT(DCode));
CREATE TABLE SECTION
	(SNum		CHAR(9)				NOT NULL,
	 Semester 	VARCHAR(10),
	 YEAR		DATE,
	 Instructer	VARCHAR(30)			NOT NULL,
	 CNum		CHAR(10)			NOT NULL,
	 PRIMARY KEY(SNum),
	 FOREIGN KEY(SNum) REFERENCES STUDENT(SNum),
	 FOREIGN KEY(CNum) REFERENCES COURSE(CNum)),
CREATE TABLE GRADE
	(NumericalGrade		INT 		NOT NULL,
	LetterGrade			CHAR(1)		NOT NULL,
	SNum		CHAR(9)				NOT NULL,
	PRIMARY KEY(LetterGrade),
	FOREIGN KEY(SNum) REFERENCES STUDENT(SNum));
